main
i_numeros, i_index, i_pivo, i_somatorio;

i_index = 0;
write("quantos numeros: ");
read(i_numeros);

repeat
write("digite um numero: ");
read(i_pivo);
i_somatorio = i_pivo;

repeat
i_pivo = i_pivo - 1;
i_somatorio = i_somatorio + i_pivo;
until i_pivo == 0;

write("somatorio: ");
write(i_somatorio);
i_index = i_index + 1;
while (i_index < i_numeros);
end
